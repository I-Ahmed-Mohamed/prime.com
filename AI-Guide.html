مثال برمجي - كيفية التكامل

يمكنك تخصيص المساعد الذكي بسهولة:

// تخصيص إعدادات المساعد
const aiSettings = {
    language: 'ar',              // اللغة: ar, en, fr, es, etc.
    personality: 'professional', // الشخصية: professional, friendly, casual
    voiceSpeed: 1.0,            // سرعة الصوت: 0.5 - 2.0
    autoTranslate: true,        // تفعيل الترجمة التلقائية
    voiceFeedback: true         // تفعيل الردود الصوتية
};

// إرسال رسالة برمجياً
window.aiAssistant.sendMessage("مرحباً!");

// الاستماع لرد المساعد
window.aiAssistant.onResponse((response) => {
    console.log('AI Response:', response);
});
